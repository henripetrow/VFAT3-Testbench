process(clk, reset) 
begin
	if rising_edge(clk) then
		if reset = '1' then
			leds(7) <= '1';
		end if;
		if fifo_in_w_en = '1' then
			leds(6) <= '1';
		end if;
		if fifo_in_r_en = '1' then
			leds(5) <= '1';
		end if;
		if fifo_in_valid = '1' then
			leds(4) <= '1';
		end if;
		if fifo_in_full = '1' then
			leds(3) <= '1';
		end if;
		if fifo_in_empty = '1' then
			leds(2) <= '1';
		end if;
		
	end if;
end process;
